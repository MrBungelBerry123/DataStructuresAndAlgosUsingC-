Reverse(Node prevNode, Node curNode):
1 - if curNode == NULL:
2 - 	root->right = prevNode
3 - 	prevNode->left = root
4 - 	return
5 - Reverse(curNode, curNode->right)
6 - curNode->right = prevNode
7 - prevNode->left = curNode
8 - prevNode->right = NULL
8 - return
